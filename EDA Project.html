import matplotlib.pyplot as plt

l = ["PM2.5", "PM10", "NO2", "SO2", "CO", "Ozone", "AQI"]

# Grouping the data by 'Days' and calculating the mean for each pollutant
df_avg = df.groupby("Days")[l].mean().reset_index()

plt.figure(figsize=(15, 10))  # Set figure size

for idx, col in enumerate(l):
    plt.subplot(3, 3, idx + 1)  # 3 rows, 3 columns of subplots
    plt.bar(df_avg["Days"], df_avg[col])  # Use the averaged data here
    plt.title(col)
    plt.xlabel("Days")
    plt.ylabel(col)

plt.tight_layout()
plt.show()
